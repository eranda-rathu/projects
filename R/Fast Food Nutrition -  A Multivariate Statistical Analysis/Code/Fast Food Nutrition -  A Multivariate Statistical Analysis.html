# removing Outliers
remove_outliers <- function(df) {
  df[] <- lapply(df, function(x) {
    if (is.numeric(x)) {
      Q1 <- quantile(x, 0.25, na.rm = TRUE)
      Q3 <- quantile(x, 0.75, na.rm = TRUE)
      IQR <- Q3 - Q1
      lower <- Q1 - 1.5 * IQR
      upper <- Q3 + 1.5 * IQR
      x[x < lower | x > upper] <- NA
    }
    x
  })
  df <- na.omit(df)
  return(df)
}
